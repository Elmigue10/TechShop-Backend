use techshop;

create table address
(
    id      int auto_increment
        primary key,
    address varchar(150) not null,
    city    varchar(30)  not null,
    region  varchar(60)  not null,
    phone   varchar(20)  not null,
    id_user int          not null
);

create table carrito
(
    id_carrito int auto_increment
        primary key,
    fecha      datetime(6) null,
    id_usuario int         null
);

create table categoria
(
    id_categoria int auto_increment
        primary key,
    categoria    varchar(255) null
);

create table detalle_carrito
(
    id_detalle_carrito int auto_increment
        primary key,
    cantidad           int null,
    id_usuario         int null,
    id_producto        int null
);

create table order_history
(
    id          int auto_increment
        primary key,
    quantity    int  not null,
    total_price int  not null,
    products    json not null
);

create table producto
(
    id           int auto_increment
        primary key,
    cantidad     int          null,
    descripcion  varchar(300) null,
    imagen_id    varchar(255) null,
    imagen_url   varchar(255) null,
    nombre       varchar(60)  null,
    precio       float        null,
    id_categoria int          null,
    constraint FKldwld2s5t6bf3wu8js8gsbgas
        foreign key (id_categoria) references categoria (id_categoria)
);

create table shipping
(
    id               int auto_increment
        primary key,
    sent_at          timestamp not null,
    received         timestamp not null,
    id_address       int       not null,
    id_order_history int       not null
);

create table shopping_cart
(
    id          int auto_increment
        primary key,
    id_product  int not null,
    id_user     int not null,
    quatity     int not null,
    total_price int not null
);

create table user
(
    id       int auto_increment
        primary key,
    name     varchar(255) null,
    password varchar(255) not null,
    username varchar(255) not null,
    email    varchar(80)  not null,
    enable   tinyint      not null,
    constraint user_password_uindex
        unique (password),
    constraint user_username_uindex
        unique (username)
);

create table user_address
(
    id         int auto_increment
        primary key,
    id_user    int not null,
    id_address int not null
);
